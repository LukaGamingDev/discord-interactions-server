/**
 * Stores and manages commands
 */
class CommandsStore {
    /**
     * @param {DiscordInteractionsServer} server
     * @param {string} [guild]
     */
    constructor(server, guild) {

        this.guild = guild
        this.cache = new Map()
    }

    }
}
